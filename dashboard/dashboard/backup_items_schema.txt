CREATE TABLE invoice_items (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    item_id TEXT,
    invoice_id TEXT,
    sku TEXT,
    name TEXT,
    quantity REAL,
    unit TEXT,
    price REAL,
    net_amount REAL,
    pre_tax_amount REAL,
    vat_amount REAL,
    discount_value REAL,
    discount_unit TEXT,
    FOREIGN KEY (invoice_id) REFERENCES invoices(id)
);
CREATE INDEX idx_item_sku ON invoice_items(sku);
CREATE INDEX idx_item_invoice_id ON invoice_items(invoice_id);
